class Solution {
    public int fib(int n) {

        int num0 = 0;
        int num1 = 1;
        if(n == 0) return num0;
        if(n == 1) return num1;

        int index = 2 , num2 = 0;
        while(index <= n){
            num2 = num0 + num1 ;
            num0 = num1;
            num1 = num2;
            index++;
        }

        return num2;

    }
}
