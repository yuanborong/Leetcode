class Solution {
    public int missingNumber(int[] nums) {

        int len = nums.length;

        int sum = 0;
        int acctual_sum = 0;
        for(int i = 0 ; i < len ; i++){
            sum += nums[i];
            acctual_sum += i;
        }
        acctual_sum += len;

        return acctual_sum - sum;

    }
}
