class Solution {
    public int findLengthOfLCIS(int[] nums) {

        int len = nums.length ;
        if(len <= 0) return 0;

        int dp = 1;
        int res = 1;
        for(int i = 1 ; i < len ; i++){
            if(nums[i] > nums[i-1]) dp = dp + 1;
            else dp = 1 ;
            if(dp > res) res = dp ;
        } 

        return res;

    }
}
