class Solution {
    public boolean canJump(int[] nums) {

        int len = nums.length;
        if(len <= 0){
            return true;
        }
        int[] dp = new int[len];
        dp[0] = nums[0];
        for(int i = 1 ; i < len ; i++){
            if(dp[i-1] == 0){
                return false;
            }
            dp[i] = (dp[i-1]-1 > nums[i])?dp[i-1]-1:nums[i];
        }
        return true;
    }
}
