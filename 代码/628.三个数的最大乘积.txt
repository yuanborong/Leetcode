class Solution {
    public int maximumProduct(int[] nums) {

        int len = nums.length;
        if(len < 3) return 0;
        if(len == 3) return nums[0] * nums[1] * nums[2];
        Arrays.sort(nums);

        int[] temp = new int[5];

        temp[0] = nums[0] * nums[1] * nums[2];
        temp[1] = nums[len-1] * nums[len-2] * nums[len-3];
        temp[2] = nums[0] * nums[1] * nums[len-1] ;

        return Math.max(Math.max(temp[0] , temp[1]) , temp[2]);

    }
}
