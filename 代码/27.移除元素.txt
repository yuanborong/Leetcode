class Solution {
    public int removeElement(int[] nums, int val) {

        int len = nums.length;
        if(len == 0){
            return 0;
        }

        int index = 0 ;
        int i;
        for(i = 0 ; i < len ; i++){
            if(nums[i] != val){
                nums[index++] = nums[i];
            }
        }

        return index;

    }
}
