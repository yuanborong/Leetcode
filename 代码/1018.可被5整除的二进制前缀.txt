class Solution {
    public List<Boolean> prefixesDivBy5(int[] A) {

        int len = A.length;
        if(len <= 0) return new ArrayList<>();

        List<Boolean> res = new ArrayList<>();
        int cur_num = 0;

        for(int i = 0 ; i < len ; i++){
            cur_num *= 2;
            cur_num += A[i] ;
            res.add((cur_num % 5 == 0) ? true : false);
            cur_num %= 5;
        }

        return res;

    }
}
