class Solution {
    public int hIndex(int[] citations) {

        int len = citations.length;
        if(len == 0){
            return 0;
        }

        Arrays.sort(citations);
        int res = Integer.MIN_VALUE;
        for(int i = 0 ; i < len ; i++){
            if(citations[i] >= (len - i) && res < (len - i)){
                res = len - i;
            }
        }
        return (res == Integer.MIN_VALUE)?0:res;

    }
}
