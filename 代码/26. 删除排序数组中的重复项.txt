class Solution {
    public int removeDuplicates(int[] nums) {

        int len = nums.length;
        if(len <= 0){
            return 0;
        }
        if(len == 1){
            return 1;
        }
        int index = 0 ; 
        int i;
        for(i = 1 ; i < len ; i++){
            if(nums[i] != nums[index]){
                nums[index+1] = nums[i];
                index++;
            }
        }
        return index + 1;

    }
}
