对字符串内容进行排序，通过先转化为 （字符数组）
char[] temp = str1.toCharArray();
然后通过Arrays类的排序功能对数组进行排序
Arrays.sort(temp);
最后把排序好的数组还原成字符串就行啦
String sorted_str = String.copyValueOf(temp);


class Solution {
    public List<List<String>> groupAnagrams(String[] strs) {

        List<List<String>> res = new ArrayList<>();        
        int len = strs.length;
        if(len <= 0){
            return res;
        }
        List<String> sorted_strs = new ArrayList<>();
        HashMap<String , List<String>> map = new HashMap<>();
        for(int i = 0 ; i < len ; i++){
            char[] temp = strs[i].toCharArray();
            Arrays.sort(temp);
            String sort_str = String.copyValueOf(temp);
            if(map.containsKey(sort_str) == false){
                List<String> new_list = new ArrayList<>();
                map.put(sort_str , new ArrayList<>());
            }
            map.get(sort_str).add(strs[i]);
        }
        for(String s : map.keySet()){
            res.add(map.get(s));
        }
        return res;
    }
}
