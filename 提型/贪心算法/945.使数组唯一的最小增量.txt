class Solution {
    public int minIncrementForUnique(int[] A) {

        int len = A.length;
        if(len <= 1) return 0;

        Arrays.sort(A);
        int res = 0;

        for(int i = 1 ; i < len ; i++){
            if(A[i] <= A[i-1]){
                res += (A[i-1] + 1 - A[i]);
                A[i] = A[i-1] + 1;
            }
        }

        return res;

    }
}
