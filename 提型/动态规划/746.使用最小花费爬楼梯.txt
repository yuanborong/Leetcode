class Solution {
    public int minCostClimbingStairs(int[] cost) {

        int len = cost.length;
        if(len <= 0) return 0;

        int[] dp = new int[len + 1];
        dp[0] = cost[0];
        dp[1] = cost[1];

        for(int i = 2 ; i < len ; i++){
            dp[i] = cost[i] + (dp[i-1] > dp[i-2] ? dp[i-2] : dp[i-1]);
        }

        return dp[len - 2] > dp[len - 1] ? dp[len - 1] : dp[len - 2];


    }
}
