class Solution {
    public double myPow(double x, int n) {

        boolean flag = false;
        long temp = n;
        if(temp == 0){
            return 1;
        }
        if(temp < 0){
            flag = true;
            temp = -temp;
        }
        double res = 1;
        while(temp > 0){
            if((temp & 1) == 1){
                res *= x;
            }
            x *= x;
            temp /= 2;
        }
        return (flag == false)?res:(1 / res);
    }
}
