class Solution {
    public int fib(int n) {
        if(n <= 0){
            return 0;
        }
        long[] dp = new long[n+1];
        dp[0] = 0 ;
        if(n == 1){
            return 1;
        }
        dp[1] = 1;
        for(int i = 2 ; i <= n ; i++){
            dp[i] = ((dp[i-1] + dp[i-2]));
            if(dp[i] >= 1000000007){
                dp[i] %= 1000000007;
            }
        }
        return (int)(dp[n]);
    }
}
