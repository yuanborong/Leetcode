class Solution {
    public int cuttingRope(int n) {

        if(n <= 0){
            return 0;
        }
        if(n == 2){
            return 1;
        }
        if(n == 3){
            return 2;
        }
        int weishu = n % 3;
        if(weishu == 0){
            return (int)Math.pow(3 , (n / 3));
        }
        else if(weishu == 1){
            return (int)Math.pow(3 , (n / 3 - 1)) * 4;
        }
        else{
            return (int)Math.pow(3 , (n / 3)) * 2;
        }

    }
}
