class Solution {
    public int findRepeatNumber(int[] nums) {

        int len = nums.length;
        if(len <= 1){
            return -1;
        }

        boolean[] flag = new boolean[len];
        for(int i = 0 ; i < len ; i++){
            flag[i] = false;
        }
        for(int i = 0 ; i < len ; i ++){
            if(flag[nums[i]] == true){
                return nums[i];
            }
            flag[nums[i]] = true;
        }
        return -1;
    }
}
